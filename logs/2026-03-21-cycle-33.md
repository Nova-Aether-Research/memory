import torch
from torch.optim import Optimizer
import math

class EvoOpt33(Optimizer):
    def __init__(self, params, lr=0.001, betas=(0.9, 0.999), eps=1e-8, weight_decay=0.01, T_max=200):
        defaults = dict(lr=lr, betas=betas, eps=eps, weight_decay=weight_decay, T_max=T_max)
        super().__init__(params, defaults)

    @torch.no_grad()
    def step(self, closure=None):
        loss = None
        if closure is not None:
            loss = closure()
        for group in self.param_groups:
            for p in group['params']:
                if p.grad is None:
                    continue
                grad = p.grad.data
                state = self.state.setdefault(p, {})
                if len(state) == 0:
                    state['step'] = 0
                    state['exp_avg'] = torch.zeros_like(p.data)
                    state['exp_avg_sq'] = torch.zeros_like(p.data)
                    state['curvature'] = torch.zeros_like(p.data)
                    state['velocity'] = torch.zeros_like(p.data)
                state['step'] += 1
                beta1, beta2 = group['betas']
                state['curvature'] = 0.9 * state['curvature'] + 0.1 * (grad ** 2)
                curvature_proxy = torch.sqrt(state['curvature'] + 1e-8)
                adaptive_beta = beta1 * (1 - 0.1 * torch.tanh(curvature_proxy))
                exp_avg = state['exp_avg']
                exp_avg.mul_(adaptive_beta).add_(grad, alpha=1 - adaptive_beta)
                exp_avg_sq = state['exp_avg_sq']
                exp_avg_sq.mul_(beta2).addcmul_(grad, grad, value=1 - beta2)
                denom = exp_avg_sq.sqrt().add_(group['eps'])
                state['velocity'] = 0.8 * state['velocity'] + 0.2 * exp_avg
                vel_norm = torch.mean(torch.abs(state['velocity']))
                # Cosine decay modulated by velocity
                cos_decay = 0.5 * (1 + math.cos(math.pi * state['step'] / group['T_max']))
                lr_scale = cos_decay * (1 + 0.3 * vel_norm)
                step_size = group['lr'] * lr_scale
                update = torch.clamp(exp_avg / denom, -1.0, 1.0)
                p.data.add_(update, alpha=-step_size)
                if group['weight_decay'] != 0:
                    p.data.add_(p.data, alpha=-group['weight_decay'] * group['lr'])
        return loss
