import torch
from torch.optim import Optimizer

class EvoOpt24(Optimizer):
    def __init__(self, params, lr=0.008, beta1=0.94, beta2_range=(0.88, 0.97), scale_amp=1.4, decay=3e-5):
        defaults = dict(lr=lr, beta1=beta1, beta2_range=beta2_range, scale_amp=scale_amp, decay=decay)
        super().__init__(params, defaults)
        self.global_step = 0
    
    @torch.no_grad()
    def step(self):
        for group in self.param_groups:
            beta2_low, beta2_high = group['beta2_range']
            for p in group['params']:
                if p.grad is None: continue
                grad = p.grad.data
                state = self.state[p]
                if len(state) == 0:
                    state['m'] = torch.zeros_like(grad)
                    state['layer_beta2'] = (beta2_low + beta2_high) / 2  # init mid
                m = state['m']
                # Evolved adaptive per-layer momentum decay
                grad_norm = grad.norm()
                adaptive_beta2 = beta2_low + (beta2_high - beta2_low) * torch.sigmoid(5 * (grad_norm - 0.1))
                state['layer_beta2'] = state['layer_beta2'] * 0.995 + adaptive_beta2 * 0.005  # slow EMA
                m.mul_(state['layer_beta2']).add_(grad, alpha=1 - state['layer_beta2'])
                scale = 1.0 + group['scale_amp'] * torch.tanh(12 * grad.abs())
                lr_t = group['lr'] / (1 + group['decay'] * self.global_step)
                p.add_(-lr_t * scale * m)
        self.global_step += 1
