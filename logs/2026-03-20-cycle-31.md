import torch
from torch.optim import Optimizer

class EvoOpt31(Optimizer):
    def __init__(self, params, lr=0.001, betas=(0.9, 0.999), eps=1e-8, weight_decay=0.0):
        defaults = dict(lr=lr, betas=betas, eps=eps, weight_decay=weight_decay)
        super().__init__(params, defaults)
        self.state = {}

    @torch.no_grad()
    def step(self, closure=None):
        loss = None
        if closure is not None:
            loss = closure()
        for group in self.param_groups:
            for p in group['params']:
                if p.grad is None:
                    continue
                grad = p.grad.data
                state = self.state.setdefault(p, {})
                if len(state) == 0:
                    state['step'] = 0
                    state['exp_avg'] = torch.zeros_like(p.data)
                    state['exp_avg_sq'] = torch.zeros_like(p.data)
                    state['curvature'] = torch.zeros_like(p.data)
                state['step'] += 1
                exp_avg, exp_avg_sq = state['exp_avg'], state['exp_avg_sq']
                beta1, beta2 = group['betas']
                # EvoOpt-31 rule: curvature proxy + tanh stabilization + dynamic clip
                state['curvature'] = 0.9 * state['curvature'] + 0.1 * (grad * grad)
                curvature_proxy = torch.sqrt(state['curvature'])
                adaptive_factor = 1 + torch.tanh(curvature_proxy * 0.1)
                # Momentum update
                exp_avg.mul_(beta1).add_(grad, alpha=1 - beta1)
                exp_avg_sq.mul_(beta2).addcmul_(grad, grad, value=1 - beta2)
                denom = exp_avg_sq.sqrt().add_(group['eps'])
                step_size = group['lr'] * adaptive_factor
                # Dynamic per-layer clip from variance history
                clip_val = torch.mean(torch.abs(exp_avg)) * 1.5
                exp_avg = torch.clamp(exp_avg, -clip_val, clip_val)
                p.data.addcdiv_(exp_avg, denom, value=-step_size)
                if group['weight_decay'] != 0:
                    p.data.add_(p.data, alpha=-group['weight_decay'] * group['lr'])
        return loss
